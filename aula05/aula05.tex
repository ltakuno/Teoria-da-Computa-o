\section{Redutibilidade}

\begin{frame} \frametitle{ Redutibilidade }
\begin{itemize}
	\item Dizemos que um problema Q reduz ao problema P se podemos usar P para resolver Q.
	\item No contexto da Teoria da computabilidade dizemos:
	
\begin{itemize}
	\item Se A for redutível a B e B for decidível, A também será decidível; e
	\item Se A for redutível a B e A for indecidível, B também será indecidível.
\end{itemize}

  \item Isto permite provar que vários problemas são indecidíveis.
\end{itemize}
\end{frame}

\begin{frame} \frametitle{ Redutibilidade }
\begin{itemize}
	\item Em outras palavras:
	
\begin{itemize}
	\item Para mostrar que um novo problema A é decidível, mostre como transformá-lo em um problema decidível conhecido B para que a solução de B possa ser usada para resolver A.
	\item Para mostrar que um novo problema A é indecidível, mostre como transformar um problema indecidível conhecido B em A para que A possa ser usado para resolver B.
\end{itemize}
	\item Redutibilidade é uma das técnicas mais importantes e mais usadas na teoria da computação.
\end{itemize}
\end{frame}

\begin{frame} \frametitle{ Teorema de Rice }
\begin{center}
\begin{tabular}{|p{9.7cm}|}
\hline
\textbf{Teorema:} Todo teste de qualquer propriedade (não trivial) de linguagens reconhecidas por máquinas de Turing é indecidível.\\
\hline
\end{tabular}
\end{center}

\textbf{Uma classe de problemas:} Seja P uma linguagem constituída de descrições de máquinas de Turing, em que P satisfaz duas condições.

\begin{enumerate}
	\item P é não-trivial: ela contém alguma descrição, mas não todas as descrições de MTs.
\begin{itemize}
	\item existe uma MT M1 para o qual $\langle M1 \rangle \in P$ , e
	\item existe uma MT M2 para o qual $\langle M2 \rangle \notin P$
\end{itemize}
	
	\item P é uma propriedade da linguagem da linguagem da MT
		$$ \mbox{se } L(M1) = L(M2) \mbox{ então } \langle M1 \rangle \in P \mbox{ sse } \langle M2 \rangle \in P$$

\end{enumerate}

\end{frame}

\begin{frame} \frametitle{ Teorema de Rice }
\begin{center}
\begin{tabular}{|p{9.7cm}|}
\hline
\textbf{Teorema:} Todo teste de qualquer propriedade (não trivial) de linguagens reconhecidas por máquinas de Turing é indecidível.\\
\hline
\end{tabular}
\end{center}

S = ``Sobre a entrada $\langle M, w \rangle$, onde M é uma MT e $w$ uma string:
\begin{enumerate}
	\item Use MT M e $w$ para construir a seguinte MT M':

	M' = Sobre a entrada $\langle T, x \rangle$, onde T é uma MT e $x$ uma string
\begin{enumerate}
	\item [1.] Simule M sobre $w$. Se pára e rejeita, \textit{rejeite}. se aceita, proceda os estágio 2
	\item [2.] Simule T sobre $x$. Se aceita, \textit{aceite} (note: $L_P$ não é trivial, então $\langle T \rangle \in L_P$ tem que existir.)
\end{enumerate}
  \item Use $M_P$ para determinar se $\langle M' \rangle \in L_P$. Se aceita, \textit{aceite}. Caso contrário, rejeite.''
\end{enumerate}
\end{frame}

\begin{frame} \frametitle{ Teorema de Rice }
A MT $M'$ simula T se M aceita $w$. Logo, $L(M')$ é igual a $L(T)$ se M aceita $w$ e é igual a 0, em caso contrário. Portanto, $\langle M, w \rangle \in P$ sse M aceita $w$.

Visto que $A_{MT}$ não é decidível, esta máquina não pode existir e nossa hipótese que $L_P$ é decidível deve ser incorreta. $\boxempty$
\end{frame}

\hline
\textbf{Lema 5.8:} Seja M um ALL com $q$ estados e $g$ símbolos no alfabeto de fita. Existem exatamente $qng^n$ configurações distintas distintas de M para uma fita de comprimente $n$.\\
\hline

O produto dessas três quantidades é o número total de configurações diferentes de M com uma fita de comprimento $n$. $\boxempty$

\end{frame}

\begin{frame} \frametitle{ Autômato linearmente limitado }
\begin{center}
\begin{tabular}{|p{9.7cm}|}

$$A_{ALL} = \{ \langle M,w \rangle | \textrm{M é um ALL que aceita a cadeia }w \}$$ é decidível.\\
\hline
\end{tabular}
\end{center}

\textbf{Prova:} O algoritmo que decide $A_{ALL}$ é como segue.

L = ``Sobre a entrada $\langle M, w \rangle$, onde M é um ALL e w é uma cadeia:
\begin{enumerate}
	\item Simule M sobre w por $qng^n$ passos ou até que ela pare.
	\item Se M parou, \textit{aceite} se ela aceitou e \textit{rejeite} se ela rejeitou. Se ela
não parou, \textit{rejeite}.''

\end{enumerate}
\end{frame}

\hline
\end{tabular}

S = ``Sobre a entrada $\langle M, w \rangle$, onde M é uma MT e w é uma cadeia:
\begin{enumerate}
	\item  Construa o ALL B a partir de $M$ e $w$ conforme descrito.
	\item  Rode R sobre a entrada $\langle B \rangle$.
	\item  Se R rejeita, aceite; se R aceita, rejeite.''
\end{enumerate} 
\end{frame}

\begin{frame} \frametitle{ Autômato linearmente limitado }

Se R aceita $\langle B \rangle$, então $L(B) = \oslash$. Então, $M$ não tem nenhuma história de computação de aceitação sobre $w$ e $M$ não aceita $w$. Conseqüentemente, S rejeita $\langle M, w \rangle$. Similarmente, se R rejeita $\langle B \rangle$, a linguagem de $B$ é não vazia. A MT B aceita uma história de computação de aceitação para $M$ sobre $w$. Portanto, M deve aceitar $w$. Como conseqüência, S aceita $\langle M, w \rangle$.

$\boxempty$
\end{frame}

\section{ Problemas indecidíveis sobre GLCs }

\begin{frame} \frametitle{ Problemas indecidíveis sobre GLCs }
\begin{center}
\begin{tabular}{|p{9.7cm}|}
\hline
\textbf{Teorema 5.10:}
$$TODAS_{GLC} = \{ \langle G \rangle | \textrm{G é uma GLC e }L(G) = \Sigma^* \}$$ é indecidível.\\
\hline
\end{tabular}
\end{center}

\textbf{Prova:} Para uma MT M e uma entrada $w$, construimos uma GLC G para gerar todas as strings que não são histórias de computação de aceitação para M sobre $w$.
\newline

Isto é, G gera todas as strings se e somente se M não aceita $w$.
\newline

Se $TODAS_{GLC}$ fosse decidível então $A_{MT}$ também seria.

\end{frame}

\begin{frame} \frametitle{ Problemas indecidíveis sobre GLCs }
\begin{center}
\begin{tabular}{|p{9.7cm}|}
\hline
\textbf{Teorema 5.10:}
$$TODAS_{GLC} = \{ \langle G \rangle | \textrm{G é uma GLC e }L(G) = \Sigma^* \}$$ é indecidível.\\
\hline
\end{tabular}
\end{center}

\textbf{Prova:} Assuma que $TODAS_{GLC}$ é decidível. Construímos uma Autômato de Pilha $D$ que aceita a string $\#C_1\#C_2^\mathcal{R}\#C_3\#C_4^\mathcal{R}\# ... \#C_l\#$, tal que $\#C_1\#C_2\#C_3\#C_4\# ... \#C_l\#$ não represente uma história de computação de aceitação de $M$ sobre $w$.
\end{frame}

\begin{frame} \frametitle{ Problemas indecidíveis sobre GLCs }
\begin{center}
\begin{tabular}{|p{9.7cm}|}
\hline
\textbf{Teorema 5.10:}
$$TODAS_{GLC} = \{ \langle G \rangle | \textrm{G é uma GLC e }L(G) = \Sigma^* \}$$ é indecidível.\\
\hline
\end{tabular}
\end{center}

D = ``Sobre a entrada $\#C_1\#C_2^\mathcal{R}\#C_3\#C_4^\mathcal{R}\# ... \#C_l\#$:
\begin{enumerate}
	\item  Se $C_1$ não é o estado inicial de $M$, então \textit{aceite}
	\item  Se $C_l$ não é o estado de aceitação de $M$, então \textit{aceite}
	\item  Se $C_i$ não produz $C_{i+1}$, então \textit{aceite}''
\end{enumerate} 

No terceiro passo, $D$ usa a pilha efetivamente. 
\end{frame}

\begin{itemize}
	\item $L(D) = \Sigma^*$  $\Leftrightarrow$ $M$ não aceita $w$, e 
	\item $L(D) \neq \Sigma^*$  $\Leftrightarrow$ $M$ aceita $w$ 
\end{itemize}
