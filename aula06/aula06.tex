\section{PCP}
\begin{frame} \frametitle{  PCP }
\begin{itemize}
	\item Dado um conjunto de ``dominós'':
	
	$\left\{ \biggl[ \dfrac{b}{ca} \biggr] , \biggl[ \dfrac{a}{ab} \biggr] , \biggl[ \dfrac{ca}{a} \biggr], \biggl[  \dfrac{abc}{c} \biggr]  \right\}$

	\item Emparelhamento:

	$ \biggl[ \dfrac{a}{ab} \biggr]$ $\biggl[ \dfrac{b}{ca} \biggr]$ $\biggl[ \dfrac{ca}{a} \biggr]$ $\biggl[ \dfrac{a}{ab} \biggr]$ $\biggl[ \dfrac{abc}{c} \biggr]$

	\item Para algumas coleções de dominós, encontrar um emparelhamento pode não ser possível:
	
	$\left\{ \biggl[ \dfrac{abc}{ab} \biggr] , \biggl[ \dfrac{ca}{a} \biggr] ,\biggl[ \dfrac{acc}{ba} \biggr] \right\}$
\end{itemize}
\end{frame}


\begin{frame} \frametitle{  PCP e PCPM }
\begin{itemize}
   \item Uma instância do PCP é uma coleção P de dominós:
   
  $ P = \left\{\biggl[ \dfrac{t_1}{b_1} \biggr], \biggl[ \dfrac{t_2}{b_2} \biggr], ..., \biggl[ \dfrac{t_k}{b_k} \biggr] \right\}$
  
  e um \textbf{emparelhamento} é uma seqüência $i_1, i_2, ..., i_l$ 
  
  onde $t_{i_1}t_{i_2} ... t_{i_l}$ = $b_{i_1}b_{i_2} ... b_{i_l}$.  
  
\begin{itemize}
	\item $ PCP = \{ \langle P \rangle |$ P é uma instância do PCP com um emparelhamento $\}$

  \item $ PCPM = \{ \langle P \rangle |$ P é uma instância do PCP com um emparelhamento que começa com o primeiro dominó $\}$

\end{itemize}
\end{itemize}
\end{frame}

\begin{frame} \frametitle{  PCP é indecidível }

\textbf{Teorema 5.15:} PCP é indecidível 

\begin{itemize}
	\item A prova envolve muitos detalhes técnicos, mas a idéia é simples.
	\item Vamos reduzir $A_{MT}$ para PCP via histórias de computação.
	\item Isto é, para uma MT M e $w$ construiremos uma instância do PCP tal que P tem uma solução se, e somente se, M aceita $w$.
\end{itemize}

\end{frame}

\begin{frame} \frametitle{  PCP é indecidível }

\textbf{Teorema 5.15:} PCP é indecidível 

\textbf{Prova:} Supomos que a MT $R$ decide o PCP e construímos S que decide $A_{MT}$. 

Seja $M = (Q, \Sigma, \Gamma, \delta, q_0, q_{accept}, q_{reject})$

Para isso, S primeiro constrói uma instância $P'$ de PCPM.

Seja $P'$ uma coleção de dominós.

\begin{enumerate}
	\item[(1)] Primeiro dominó em $P'$
	
	$\biggl[ \dfrac{\#}{\#q_0w_1w_2 ... w_n\#} \biggr]$
	
	\item[(2)] Para todo $a, b \in \Gamma$ e todo $q, r \in Q$ 
	
	se $\delta(q,a) = (r,b,D)$, $\biggl[ \dfrac{qa}{br} \biggr] \in P'$
	
	
\end{enumerate}
\end{frame}


\begin{frame} \frametitle{  PCP é indecidível }

\textbf{Teorema 5.15:} PCP é indecidível 

\textbf{Prova: (Continuação)} 

\begin{enumerate}
	\item[(3)] Para todo $a, b, c \in \Gamma$ e todo $q, r \in Q$ 
	
	se $\delta(q,a) = (r,b,E)$, $\biggl[ \dfrac{cqa}{rcb} \biggr] \in P'$

\textbf{Teorema 5.15:} PCP é indecidível 

\textbf{Prova: (Continuação)}

\section{Redutibilidade por Mapeamento}
\begin{frame} \frametitle{  Redutibilidade }

Suponha que nós temos dois problema A e B. Rigorosamente, se a solução de B pode ser usado para resolver o problema de A então dizemos que:
\begin{block}{}
\begin{center}
A é redutível a B.
\end{center}
\end{block}
\end{frame}

\begin{frame} \frametitle{  Redutibilidade }
\begin{itemize}
	\item O problema de descobrir como sair do ponto A e chegar no ponto B em São Paulo pode ser reduzido a tarefa de encontrar um mapa para a cidade de São Paulo. 
\end{itemize}
\end{frame}

\begin{frame} \frametitle{  Redutibilidade }
\begin{itemize}
	\item O problema de atravessar um rio pode ser reduzido ao encontrar um barco. 
\end{itemize}
\end{frame}

\begin{frame} \frametitle{  Redutibilidade }
\begin{itemize}
	\item O problema de saciar a sede pode ser reduzido ao problema de encontrar água limpa. 
\end{itemize}
\end{frame}

\begin{frame} \frametitle{  Redutibilidade }
\begin{itemize}
	\item O problema de resolver uma equação quadrática pode ser reduzido ao problema de fazer amigos matemáticos. 
\end{itemize}
\end{frame}

\begin{frame} \frametitle{  Redutibilidade por Mapeamento }
\begin{center}

$$w \in A \Longleftrightarrow f(w) \in B$$

A função f é denominada a \textbf{redução} de A para B.

\\

\end{tabular}

\begin{frame} \frametitle{  Redutibilidade por Mapeamento }
\begin{center}
\begin{tabular}{|p{9.0cm}|}
\hline
\textbf{Teorema 5.22: } Se $A \leq_m B$ e B é decidível, então A é decidível.\\
\hline
\end{tabular}
\end{center}

\textbf{Prova:} Fazemos $M$ ser o decisor para $B$ e $f$ a redução de $A$ para $B$. Descrevemos um decisor $N$ para $A$ da seguinte forma.


$N$ = ``Sobre a entrada $w$:
\begin{enumerate}
	\item Compute $f(w)$.
	\item Rode $M$ sobre a entrada $f(w)$ e dê como o que $M$ der como saída.
\end{enumerate}
Claramente, se  $w \in A$, então $f(w) \in B$, porque $f$ é uma redução de $A$ para $B$. Portanto, $M$ aceita $f(w)$ sempre que $w \in A$.

\end{frame}


\begin{frame} \frametitle{  Redutibilidade por Mapeamento }

\begin{center}
\begin{tabular}{|p{9.0cm}|}
\hline
\textbf{Corolário 5.23:} Se $A \leq_m B$ e A é indecidível, então B é indecidível.\\
\hline
\end{tabular}
\end{center}

\textbf{Prova:} Assuma que $B$ é decidível, seja M um decisor para $B$ e seja $f$ uma redução de $A$ para $B$, então podemos construir um decisor $N$ para $A$ da seguinte forma.


$N$ = ``Sobre a entrada $w$:
\begin{enumerate}
	\item Compute $f(w)$.
	\item Rode $M$ sobre a entrada $f(w)$ e dê como o que $M$ der como saída.
\end{enumerate}
Claramente, como $A$ é indecidível esta máquina não pode existir, portanto, nossa suposição de que $B$ é decidível deve estar incorreta.
\end{frame}


\begin{frame} \frametitle{ Exemplo $A_{MT} \leq_m PARA_{MT}$}
\begin{center}
\begin{tabular}{|p{10.0cm}|}
\hline
Seja \\

\hspace{0.5cm} $PARA_{MT}$ = $\{ \langle M, w \rangle | M$ é uma MT para sobre $w$ $\}$ \\\\
\hline
\end{tabular}
\end{center}



Para mostrar que 
$$A_{MT} \leq_m PARA_{MT}$$

temos que encontrar uma função computável tal que:

$$f(\langle M,w \rangle) = \langle M',w' \rangle$$

tal que :

$$\langle M,w \rangle \in A_{MT}  \mbox{ se e somente se } \langle M' ,w' \rangle \in PARA_{MT}$$
\end{frame}


\begin{frame} \frametitle{ Exemplo $A_{MT} \leq_m PARA_{MT}$}
Para construir a função computável $f$, construa uma MT F que computa a redução:

F = ``Sobre a entrada $\langle M, w \rangle$:

\begin{enumerate}
\item Construa a seguinte máquina $M'$

$M'$ = ``Sobre a entrada x:
	\begin{enumerate}
	\item Rode M sobre x
	\item Se M aceitar, aceite.
	\item Se M rejeitar, entrar em loop."
	\end{enumerate}
\item Dê como saída $\langle M', w \rangle$
\end{enumerate}

Observe que $\langle M', w \rangle$ sse $\langle M, w \rangle \in A_{MT}$ como requerido.
\end{frame}

\begin{frame} \frametitle{ Exemplo $A_{MT} \leq_m PARA_{MT}$}
Para construir a função computável $f$, construa uma MT F que computa a redução:

F = ``Sobre a entrada $\langle M, w \rangle$:

\begin{enumerate}
\item Construa a seguinte máquina $M'$

$M'$ = ``Sobre a entrada x:
	\begin{enumerate}
	\item Rode M sobre x
	\item Se M aceitar, aceite.
	\item Se M rejeitar, entrar em loop."
	\end{enumerate}
\item Dê como saída $\langle M', w \rangle$
\end{enumerate}

Se a entrada fornecida para F não é um elemento de $A_{MT}$, assumimos que F mapeia para alguma string que não esteja em $PARA_{MT}$.
\end{frame}

\begin{frame} \frametitle{ Exemplo $A_{MT} \leq_m PARA_{MT}$}

Vamos analisar por que isto funciona:

\begin{itemize}
\item $\langle M, w \rangle \in A_{MT} \rightarrow M$ aceita $w$ $\rightarrow M'$ aceita $w \rightarrow \langle M', w \rangle \in PARA_{MT}$ 

\item $\langle M, w \rangle \notin A_{MT}$
\begin{itemize}
\item $M$ rejeita $w  \rightarrow M'$ entra em loop $ \rightarrow \langle M', w \rangle \notin PARA_{MT}$
\item $M$ entra em loop sobre $w \rightarrow M'$ entra em loop $ \rightarrow \langle M', w \rangle \notin PARA_{MT}$ 
\end{itemize}
\end{itemize}

Codificação incorreta de $\langle M, w \rangle$
\begin{itemize}
\item $\langle M, w \rangle \notin A_{MT}$
\item Faça $f(\langle M, w \rangle) = \langle M, w \rangle \notin PARA_{MT}$
\end{itemize}
\end{frame}


\begin{frame} \frametitle{  Exemplo: $A_{MT} \leq_m PCPM \leq_m PCP $ }
Na prova da indecidibilidade do problema da correspondência de Post mostramos duas reduções por mapeamento:

$$A_{MT} \leq_m PCPM$$

$$PCPM \leq_m PCP$$

Note que as reduções por mapeamento são transitivas.
\end{frame}



\begin{frame} \frametitle{  Exemplo: $V_{MT} \leq_m EQ_{MT}$ }
\begin{center}
\begin{tabular}{|p{9.7cm}|}
\hline
 A linguagem $$V_{MT} = \{ \langle M \rangle | \textrm{M é uma MT L(M) = }\oslash \}$$ é indecidível.\\
\hline
\end{tabular}
\end{center}

Vamos mostrar que 
$$V_{MT} \leq_m EQ_{MT}$$

Seja $M_1$ uma MT que rejeita todas as entradas. 

Defina:

$$f(\langle M \rangle) = \langle M,M_1 \rangle$$

então teremos

\begin{center}
$L(M) = \oslash$ se e somente se $L(M) = L(M_1)$
\end{center}
\end{frame}


\begin{frame} \frametitle{  Exemplo: $A_{MT} \leq_m \overline{V_{MT}}$ }
Vamos mostrar que 
$$A_{MT} \leq_m \overline{V_{MT}}$$

Defina:

$$f(\langle M,w \rangle) = \langle M_1 \rangle$$

tal que 
\begin{center}
$M$ aceita $w$ se e somente se $L(M_1) \neq \oslash$
\end{center}

\end{frame}

\begin{frame} \frametitle{   $A_{MT} \leq_m \overline{V_{MT}}$ }
Para construir a função computável $f$, construa uma MT F que computa a redução:

\begin{enumerate}
	\item Sobre a entrada $\langle M,w\rangle$
  \item Seja $M_1$ a máquina:
\begin{enumerate}
  \item    ``Sobre a entrada x
  \item     se $x \neq w$ rejeite
  \item     senão execute M sobre x
  \item     se M aceita x, aceite''
\end{enumerate}
  \item Dê como saida $\langle M_1 \rangle$
\end{enumerate}\end{frame}

\begin{frame} \frametitle{   $A_{MT} \leq_m \overline{V_{MT}}$ }
Vamos analisar por que isto funciona:

\begin{itemize}
\item $\langle M,w\rangle \in A_{MT} \rightarrow L(M_1) = \{w\} \rightarrow \langle M_1\rangle \notin V_{MT} \rightarrow \langle M_1\rangle \in \overline{V_{MT}}$
\item $\langle M,w\rangle \notin A_{MT} \rightarrow L(M_1) = \oslash \rightarrow \langle M_1\rangle \in V_{MT} \rightarrow \langle M_1\rangle \notin \overline{V_{MT}}$
\end{itemize}

\end{frame}

\begin{frame} \frametitle{   $A_{MT} \leq_m \overline{V_{MT}}$ }
A redutibilidade anterior mostra que 
\begin{itemize}
	\item $\overline{V_{MT}}$ é indecidível
\end{itemize}

isto implica que 
\begin{itemize}
	\item $V_{MT}$ é indecidível
\end{itemize}

Conclusão:

Para mostrar que uma linguagem B é indecidível.
\begin{itemize}
\item Encontre uma linguagem $A$ que foi provado ser indecidível.
\item Encontre uma função de redução de $A$ para $B$ ou para $\overline{B}$
\end{itemize}
\end{frame}

\begin{frame} \frametitle{  Redutibilidade por Mapeamento }
\textbf{Observação:} Reduções de linguagens nem sempre existem. Isto é, nem sempre é possível especificar uma função computação que reduz uma linguagem na outra. 

\vspace{1cm} 

Ex: $A_{MT} \leq_m V_{MT}$

\vspace{1cm} 
Não é possível construir uma redução, como pede o exercício 5.5 do livro.

\end{frame}

\begin{frame} \frametitle{  Redutibilidade por Mapeamento }
\begin{center}
\begin{tabular}{|p{9.0cm}|}
\hline
\textbf{Teorema 5.28: } Se $A \leq_m B$ e B é Turing-reconhecível, então A é Turing-reconhecível.\\
\hline
\end{tabular}
\end{center}
\textbf{Prova:} Seja M um reconhecedor para B e $f$ é uma redução de A para B. A seguinte MT N reconhece A:

N = ``Sobre a entrada $w$
\begin{enumerate}
	\item Compute $f(w)$
	\item Execute M sobre $f(w)$ e dê como saída o que M apresentar.
\end{enumerate}

Claramente, se $w \in A$ então $f(w) \in B$ pois $f$ é uma redução. Assim $M$ aceita $f(w)$ quando $w \in A$.
\end{frame}

\textbf{Questão:}Qual linguagem estudada não é Turing-Reconhecível ?

\textbf{Resposta:} $\overline{A_{MT}}$

\begin{frame} \frametitle{ $EQ_{MT}$ não é Turing-reconhecível }
$A_{MT} \leq_m \overline{EQ_{MT}}$.


Vamos analisar por que isto funciona:

\begin{itemize}
\item $\langle M, w \rangle \in A_{MT} \rightarrow M$ aceita $w \rightarrow $ $L(M_2) = \Sigma^* \rightarrow$ $L(M_1) \neq L(M_2) \rightarrow$ $\langle M_1, M_2\rangle \in \overline{EQ_{MT}}$ 

\item $\langle M, w \rangle \notin A_{MT} \rightarrow M$ não aceita $w \rightarrow $ $L(M_2) = \oslash \rightarrow$ $L(M_1)= L(M_2) \rightarrow$ $\langle M_1, M_2\rangle \notin \overline{EQ_{MT}}$ 

\end{itemize}
\end{frame}

\begin{itemize}

\begin{frame} \frametitle{ }

A única diferença entre $f$ e $g$ está na máquina $M_1$. Em $f$, a máquina $M_1$ sempre rejeita enquento em $g$ ela sempre aceita. Em ambas $f$ e $g$, $M$ aceita $w$ sse $M_2$ sempre aceita. 
\end{frame}
